<CSFrame>
    <CSTitle>Factorial</CSTitle>
    <GatherLatex code="
        n!
        = \prod_{k = 1}^n{k}
        = 1 \times 2 \times 3 \times \dots \times (n-2) \times (n-1) \times n,
        \\
        \forall n \in \mathbb{Z} : n > 0.
        \qquad\qquad
        0! = 1
    " />

    Recursive definition:
    <DisplayLatex code="
        0! = 1
        \qquad\qquad
        n! = n \parens{n - 1} !
    " />

    <Todo>Binet's formula?</Todo>
</CSFrame>

<CSFrame>
    <CSTitle><Latex code="k" />-Permutations of <Latex code="n" /></CSTitle>
    <DisplayLatex code="
        \MyPermutations{n}{k}
        = \frac{n!}{\parens{n - k}!}
    " />

    <CSTitle><Latex code="k" />-Combinations of <Latex code="n" /> / Binomial Coefficient</CSTitle>
    <DisplayLatex code="
        \MyCombinations{n}{k}
        = \binom{n}{k}
        = \frac{n!}{k! \, \parens{n - k}!}
    " />
</CSFrame>

    <CSTitle>Distributive Laws</CSTitle>
    <GatherLatex code="
        A \cdot (B + C) = (A \cdot B) + (A \cdot C) \\
        A + (B \cdot C) = (A + B) \cdot (A + C)
    " />

    <CSTitle>Absorption Laws</CSTitle>
    <GatherLatex code="
        A + (A \cdot B) = A \\
        A \cdot (A + B) = A \\
        (A \cdot B) + (A \cdot \overline{B}) = A \\
        (A + B) \cdot (A + \overline{B}) = A
    " />

    <CSTitle>De Morgan's Theorem</CSTitle>
    <GatherLatex code="
        \overline{A + B} = \overline{A} \cdot \overline{B} \\
        \overline{A \cdot B} = \overline{A} + \overline{B}
    " />
</CSFrame>

<CSFrame>
    <CSTitle>Modular Arithmetic Properties</CSTitle>
    <AlignLatex code="
        \parens{a + b} \bmod m
            &= \parens{\parens{a \bmod m} + \parens{b \bmod m}} \bmod m
            \\
        \parens{a - b} \bmod m
            &= \parens{\parens{a \bmod m} - \parens{b \bmod m}} \bmod m
            \\
        \parens{a \times b} \bmod m
            &= \parens{\parens{a \bmod m} \parens{b \bmod m}} \bmod m
    " />
</CSFrame>

</CheatsheetBlock>
