\newcommand{\SimpleVennDiagram}[3]{%
    \begin{tikzpicture}[scale=0.4, transform shape]
        \begin{scope} % Left
            \clip
                (-1, -1) rectangle (2.2, 1)
                (1.2, 0) circle (1)
            ;
            \fill[#1] (0, 0) circle (1);
        \end{scope}
        \begin{scope} % Middle
            \clip (1.2, 0) circle (1);
            \fill[#2] (0, 0) circle (1);
        \end{scope}
        \begin{scope} % Right
            \clip
                (-1, -1) rectangle (2.2, 1)
                (0, 0) circle (1)
            ;
            \fill[#3] (1.2, 0) circle (1);
        \end{scope}
        \begin{scope} % Outline
            \draw[line width=2.0pt]
                (1.2, 0) circle (1)
                (0, 0) circle (1)
            ;
        \end{scope}
    \end{tikzpicture}%
}%
