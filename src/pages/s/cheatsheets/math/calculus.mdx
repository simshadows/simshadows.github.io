    <CSTitle>Chain Rule</CSTitle>
    <DisplayLatex code="
        \frac{\memphRC{\diff{y}}}{\memphG{\diff{x}}}
        = \frac{\memphRC{\diff{y}}}{\memphBC{\diff{u}}}
        \times \frac{\memphBC{\diff{u}}}{\memphG{\diff{x}}}
    " />

    <CSTitle>Product Rule</CSTitle>
    <GatherLatex code="
        \frac{\diff{}}{\diff{x}} \parens{\memphRC{u} \memphG{v}}
        = \memphRC{u} \memphG{ \frac{\diff{v}}{\diff{x}}}
        + \memphG{v}  \memphRC{\frac{\diff{u}}{\diff{x}}}
        \\
        \frac{\diff{}}{\diff{x}} \parens{\memphRC{u} \memphG{v} \memphBC{w}}
        = \memphG{v}  \memphBC{w} \memphRC{\frac{\diff{u}}{\diff{x}}}
        + \memphRC{u} \memphBC{w} \memphG{ \frac{\diff{v}}{\diff{x}}}
        + \memphRC{u} \memphG{v}  \memphBC{\frac{\diff{w}}{\diff{x}}}
    " />

    <CSTitle>Quotient Rule</CSTitle>

            \memphBC{u} \memphRC{a'} - \memphRC{a} \memphBC{u'}

        }
    " />
</CSFrame>

<CSFrame>
    <CSTitle>Notation of Second Order Partial Derivatives</CSTitle>

    Consider a function of two variables <Latex code="f(x, y)" />.

    <AlignatLatex
        eqcols={4}
        moreMacros={{
            "\\Fun":    "\\memphG{f}",
            "\\X":      "\\memphBC{x}",
            "\\Y":      "\\memphRC{y}",
            "\\DFun":   "\\memphG{\\partial{f}}",
            "\\DX":     "\\memphBC{\\partial{x}}",
            "\\DY":     "\\memphRC{\\partial{y}}",
            "\\DsqFun": "\\memphG{\\partial^2{f}}",
            "\\DXsq":   "\\memphBC{\\partial{x}^2}",
            "\\DYsq":   "\\memphRC{\\partial{y}^2}"
        }}
        code="
            \frac{\DsqFun}{\DXsq}
                &= \frac{\partial{}}{\DX} && \parens{\frac{\DFun}{\DX}} &
                &= (\memphG{f}_{\X})_{\X} &
                &= \memphG{f}_{\X \X}
                \\
            \frac{\DsqFun}{\DYsq}
                &= \frac{\partial{}}{\DY} && \parens{\frac{\DFun}{\DY}} &
                &= (\memphG{f}_{\Y})_{\Y} &
                &= \memphG{f}_{\Y \Y}
                \\[2em]
            \frac{\DsqFun}{\DY \, \DX}
                &= \frac{\partial{}}{\DY} && \parens{\frac{\DFun}{\DX}} &
                &= (\memphG{f}_{\X})_{\Y} &
                &= \memphG{f}_{\X \Y}
                \\
            \frac{\DsqFun}{\DX \, \DY}
                &= \frac{\partial{}}{\DX} && \parens{\frac{\DFun}{\DY}} &
                &= (\memphG{f}_{\Y})_{\X} &
                &= \memphG{f}_{\Y \X}
        "
    />
</CSFrame>

    Consider a function of two variables <Latex code="f(x, y)" />.
    
    If <Latex code="f" /> and all of its first and second order partial derivatives are continuous, then:

    <DisplayLatex

            \frac{\DsqFun}{\DX \, \DY}

    <CSTitle>Integration By Substitution</CSTitle>
    <DisplayLatex code="
        \memphRC{u} \defeq \memphRC{\varphi(x)}
        \\
        \begin{aligned}
            \int_a^b{\memphBC{f(x)} \,\diff{x}}
            &= \int_a^b{
                \memphRC{\varphi'(x)}
                \, \memphBC{f(\memphRC{\varphi(x)})}
                \, \diff{x}
            }
            \\
            &= \int_a^b{
                \memphRC{\frac{\diff{u}}{\diff{x}}}
                \, \memphBC{f(\memphRC{u})}
                \, \diff{x}
            }
            = \int_{\memphRC{\varphi(a)}}^{\memphRC{\varphi(b)}}{
                \, \memphBC{f(\memphRC{u})}
                \, \memphRC{\diff{u}}
            }
        \end{aligned}
    " />
