    <CSTitle>Factorial</CSTitle>
    <GatherLatex code="
        n!
        = \prod_{k = 1}^n{k}
        = 1 \times 2 \times 3 \times \dots \times (n-2) \times (n-1) \times n,
        \\
        \forall n \in \mathbb{Z} : n > 0.
        \qquad\qquad
        0! = 1
    " />
</CSFrame>

    <CSTitle><Latex code="k" />-Permutations of <Latex code="n" /></CSTitle>

        = \frac{n!}{\parens{n - k}!}
    " />

    <CSTitle><Latex code="k" />-Combinations of <Latex code="n" /> / Binomial Coefficient</CSTitle>
    <DisplayLatex code="
        \MyCombinations{n}{k}
        = \binom{n}{k}
        = \frac{n!}{k! \, \parens{n - k}!}
    " />
