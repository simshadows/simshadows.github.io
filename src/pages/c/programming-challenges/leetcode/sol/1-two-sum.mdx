```py
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        for i, iv in enumerate(nums):
            for j, jv in enumerate(nums[i + 1:]):
                if iv + jv == target:
                    return [i, i + j + 1]
        raise RuntimeError()
```
