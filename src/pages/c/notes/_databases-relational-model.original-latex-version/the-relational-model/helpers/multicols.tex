\def\MulticolsReset{%
    \setlength{\columnseprule}{0pt}%
    \setlength{\columnsep}{10pt}%
    \def\columnseprulecolor{\normalcolor}%
}
\MulticolsReset

\newcommand{\MulticolsBreak}{\vfill\null\columnbreak} % For use in the multicols environment.
\newcommand{\MulticolsCleanEnd}{\vfill\null} % For use in the multicols environment.
\newcommand{\MulticolsPhantomPlaceholder}{\phantom{x}} % An arbitrary box to occupy a multicols column.

\newenvironment{MulticolsSoftSepRule}[1]{%
    \setlength{\columnseprule}{0.4pt}%
    \setlength{\columnsep}{20pt}%
    \def\columnseprulecolor{\color{lightgray}}%
    \begin{multicols}{#1}%
}{%
    \MulticolsCleanEnd%
    \end{multicols}%
    \MulticolsReset%
}
